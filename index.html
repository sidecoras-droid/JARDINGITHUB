    <script>

        function handleGlobalClick(event) {

            if (event.target.id === 'workspace') {

                const text = document.getElementById('jardin-text');

                text.style.display = 'block';

            }

        }


        function toggleOverlay(id, event) {

            if (event) event.stopPropagation();

            document.querySelectorAll('.overlay').forEach(ov => {

                if (ov.id !== id) ov.style.display = 'none';

            });

            const overlay = document.getElementById(id);

            const isVisible = (overlay.style.display === 'flex');

            overlay.style.display = isVisible ? 'none' : 'flex';

        }


        function showHomeText(event) {

            if (event) event.stopPropagation();

            document.querySelectorAll('.overlay').forEach(ov => ov.style.display = 'none');

            const text = document.getElementById('jardin-text');

            text.style.display = 'block';

        }


        function validateAndSend() {

            const emailInput = document.getElementById('email');

            const email = emailInput.value;

            const message = document.getElementById('form-message');

            

            // Regex para proveedores reconocidos (gmail, outlook, hotmail, yahoo, icloud)

            const recognizedProviders = /@(gmail|outlook|hotmail|yahoo|icloud)\./i;

            const emailRegex = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;


            if (!emailRegex.test(email)) {

                message.style.color = "#d00";

                message.innerText = "Dirección de correo no válida.";

                return;

            }


            if (!recognizedProviders.test(email)) {

                message.style.color = "#d00";

                message.innerText = "Por favor, usa un proveedor reconocido (Gmail, Outlook, etc.).";

                return;

            }


            message.style.color = "green";

            message.innerText = "Formulario enviado correctamente.";

        }


        function updateMap() {

            const address = document.getElementById('addressInput').value;

            if (address) {

                const mapIframe = document.getElementById('googleMap');

                const encodedAddress = encodeURIComponent(address);

                mapIframe.src = `https://maps.google.com/maps?q=${encodedAddress}&t=&z=13&ie=UTF8&iwloc=&output=embed`;

            }

        }

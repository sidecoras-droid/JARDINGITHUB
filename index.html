    <script>

        /**

         * Maneja clics en el área principal para mostrar el nombre del jardín

         */

        function handleGlobalClick(event) {

            if (event.target.id === 'workspace') {

                const text = document.getElementById('jardin-text');

                text.style.display = 'block';

            }

        }


        /**

         * Alterna la visibilidad de las pantallas superpuestas (overlays)

         * @param {string} id - ID del elemento overlay a mostrar/ocultar

         * @param {Event} event - Evento del ratón para detener la propagación

         */

        function toggleOverlay(id, event) {

            if (event) event.stopPropagation();

            

            // Cierra cualquier otro overlay que esté abierto

            document.querySelectorAll('.overlay').forEach(ov => {

                if (ov.id !== id) ov.style.display = 'none';

            });


            const overlay = document.getElementById(id);

            const isVisible = (overlay.style.display === 'flex');

            overlay.style.display = isVisible ? 'none' : 'flex';

        }


        /**

         * Cierra overlays y muestra el texto central desde el icono Home

         */

        function showHomeText(event) {

            if (event) event.stopPropagation();

            document.querySelectorAll('.overlay').forEach(ov => ov.style.display = 'none');

            const text = document.getElementById('jardin-text');

            text.style.display = 'block';

        }

    </script>
